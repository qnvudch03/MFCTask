#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> answer;

for (int i = 0; i < arr.size(); i++)
{
    int targetNum = arr[i];

    if (flag[i] == true)
    {
        for (int n = 0; n < targetNum * 2; n++)
            answer.push_back(targetNum);
    }

    else
    {
        for (int n = 0; n < targetNum; n++)
            answer.pop_back();
    }
}

return answer;

}
