#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    int width, height = 0;
    
    for(int i = 0; i < dots.size() - 1; i++)
    {
        vector<int> pos1 = dots[i];
        vector<int> pos2 = dots[i+1];
        
        if(pos1[0] != pos2[0] && width == 0)
            width = pos1[0] -pos2[0];
        
        if(pos1[1] != pos2[1] && height == 0)
            height = pos1[1] -pos2[1];
    }
    
    answer = width * height;
    
    return (answer > 0) ? answer : -answer;
}
